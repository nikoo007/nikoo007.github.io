import {getUserProfile} from '@/api/user.js'
export default {
  name: 'userProfile',
  data() {
    return {
      user: {},
    }
  },

  created() {
    this.loadProfile()
  },

  methods: {
    async loadProfile() {
      try {
        const res = await getUserProfile()
        this.user = res.data.data
      } catch (err) {
        this.$toast('获取数据失败')
      }
    },
  },
}
